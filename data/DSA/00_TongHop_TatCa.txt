#Q42 [SC4]
Mỗi nút danh sách liên kết đơn: struct Node { int id; Node* next; }
Node* proc(Node* f, int v) {
  if (f == NULL) return NULL;
  if (f->id == v) { Node* tm=f; f=f->next; free(tm); return f; }
  f->next = proc(f->next, v);
  return f;
}
Hàm proc thực hiện công việc gì?
A. Loại bỏ phần tử ĐẦU TIÊN có giá trị bằng v
B. Loại bỏ phần tử CUỐI CÙNG có giá trị bằng v
C. Loại bỏ TẤT CẢ phần tử có giá trị bằng v
D. Không thay đổi danh sách
#ANSWER: A
#EXPLAIN: Hàm duyệt từ đầu, ngay khi tìm thấy id==v thì giải phóng và return ngay, không xét tiếp phần còn lại.

#Q43 [SC4]
Cùng cấu trúc Node như trên.
Node* proc(Node* f, int v) {
  if (f == NULL) return NULL;
  if (f->id == v) { Node* tm=f; f=f->next; free(tm); f=proc(f,v); return f; }
  f->next = proc(f->next, v);
  return f;
}
Hàm proc thực hiện công việc gì?
A. Loại bỏ phần tử đầu tiên có giá trị bằng v
B. Loại bỏ TẤT CẢ phần tử có giá trị bằng v
C. Đảo ngược danh sách
D. Không thay đổi danh sách
#ANSWER: B
#EXPLAIN: Sau khi free(tm), hàm tiếp tục gọi đệ quy proc(f,v) cho phần còn lại nên xóa hết mọi phần tử bằng v.

#Q44 [SC4]
Node* makeNode(int v) { Node* p=malloc(sizeof(Node)); p->id=v; p->next=NULL; return p; }
Node* proc(Node* f, int v) {
  if (f == NULL) return makeNode(v);
  Node* p = makeNode(v);
  p->next = f;
  return p;
}
Hàm proc thực hiện công việc gì?
A. Thêm nút có giá trị v vào ĐẦU danh sách
B. Thêm nút có giá trị v vào CUỐI danh sách
C. Xóa nút đầu danh sách
D. Đảo ngược danh sách
#ANSWER: A
#EXPLAIN: Luôn tạo nút mới p, gán p->next = f, làm p trở thành nút đầu tiên của danh sách.

Node* proc(Node* f, int v) {
  if (f == NULL) return makeNode(v);
  f->next = proc(f->next, v);
  return f;
}
Hàm proc thực hiện công việc gì?

#ANSWER: B
#EXPLAIN: Đệ quy đi đến hết danh sách (f==NULL) rồi mới tạo nút mới và nối vào, nên nút mới nằm ở cuối.

#Q47 [SC4]
Node* proc(Node* f) {
  Node *p=f, *pp=NULL, *np=NULL;
  while (p != NULL) {
    np = p->next; p->next = pp; pp = p; p = np;
  }
  return pp;
}
Hàm proc thực hiện công việc gì?
A. Đảo ngược (Reverse) danh sách liên kết
B. Sắp xếp danh sách liên kết

D. Nối hai danh sách liên kết
#ANSWER: A
#EXPLAIN: Dùng 3 con trỏ để lần lượt đổi hướng con trỏ next về phía trước, đảo ngược toàn bộ danh sách.
